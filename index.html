#include <SDL2/SDL.h>
#include <math.h>

    void drawCircle(SDL_Renderer *renderer, int cx, int cy, int radius) {
        for (int w = 0; w < radius * 2; w++) {
            for (int h = 0; h < radius * 2; h++) {
                int dx = radius - w;
                int dy = radius - h;
                if ((dx*dx + dy*dy) <= (radius * radius)) {
                    SDL_RenderDrawPoint(renderer, cx + dx, cy + dy);
                }
            }
        }
    }

    int main() {
        SDL_Init(SDL_INIT_VIDEO);

        SDL_Window *window = SDL_CreateWindow(
            "SDL Circle",
            SDL_WINDOWPOS_CENTERED,
            SDL_WINDOWPOS_CENTERED,
            500, 400, 0
        );

        SDL_Renderer *renderer = SDL_CreateRenderer(window, -1, 0);

        // background black
        SDL_SetRenderDrawColor(renderer, 0, 0, 0, 255);
        SDL_RenderClear(renderer);

        // circle color white
        SDL_SetRenderDrawColor(renderer, 255, 0, 255, 255);

        drawCircle(renderer, 250, 200, 40);

        SDL_RenderPresent(renderer);

        SDL_Event e;
        int running = 1;
        while (running) {
            while (SDL_PollEvent(&e)) {
                if (e.type == SDL_QUIT)
                    running = 0;
            }
        }

        SDL_DestroyRenderer(renderer);
        SDL_DestroyWindow(window);
        SDL_Quit();
    }

Bresenham Line Algorithm

The main goal is to draw a straight line on a pixel screen (like a computer display) using only integer calculations ‚Äî fast and efficient. Bresenham‚Äôs algorithm helps choose the best next pixel at each step.

Core Idea

When drawing a line from left to right on a pixel screen, the computer cannot draw a perfect continuous line. It must choose the nearest pixel at each step.

At each step in x, there are only two possible pixel choices:

  • Go straight ‚Üí (x+1, y)
  • Go up ‚Üí (x+1, y+1)

Bresenham‚Äôs algorithm checks which of these two pixels is closer to the actual mathematical line and selects that pixel.

Instead of using slow floating-point calculations, it uses a small integer decision value to choose the next pixel ‚Äî which makes the algorithm very fast and efficient. 

#include <SDL2/SDL.h>
#include <stdlib.h>

void drawLine(SDL_Renderer* r, int x1, int y1, int x2, int y2) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);
    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;
    int err = dx - dy;

    while (1) {
        SDL_RenderDrawPoint(r, x1, y1);
        if (x1 == x2 && y1 == y2) break;

        int e2 = 2 * err;
        if (e2 > -dy) { err -= dy; x1 += sx; }
        if (e2 < dx) { err += dx; y1 += sy; }
    }
}

int main() {
    SDL_Init(SDL_INIT_VIDEO);

    SDL_Window* win = SDL_CreateWindow(
        "Bresenham Line",
        SDL_WINDOWPOS_CENTERED,
        SDL_WINDOWPOS_CENTERED,
        600, 400, 0
    );

    SDL_Renderer* ren = SDL_CreateRenderer(win, -1, 0);

    SDL_SetRenderDrawColor(ren, 0, 0, 0, 255);
    SDL_RenderClear(ren);

    SDL_SetRenderDrawColor(ren, 255, 255, 0, 255);

    drawLine(ren, 50, 50, 500, 300);

    SDL_RenderPresent(ren);

    SDL_Event e;
    int running = 1;
    while (running) {
        while (SDL_PollEvent(&e)) {
            if (e.type == SDL_QUIT)
                running = 0;
        }
    }

    SDL_DestroyRenderer(ren);
    SDL_DestroyWindow(win);
    SDL_Quit();
}
